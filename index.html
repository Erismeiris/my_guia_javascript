Vicular el archivo javascript al archivo .html

Para vincular un archivo javascript a un archivo html se debe hacer uso de la etiqueta script, la cual se puede colocar en el head o en el body del archivo html. A continuación se muestra un ejemplo de cómo vincular un archivo javascript al archivo html.

        
            <html>
                <head>
                    <title>Título de la página</title>
                    <script src="ruta-del-archivo.js"></script>
                </head>
                <body>
                    <h1>Hola mundo</h1>
                </body>
            </html>

Callback

Un callback es una función que se pasa como argumento a otra función y se ejecuta después de que la función principal ha terminado de ejecutarse. A continuación se muestra un ejemplo de cómo utilizar un callback en javascript.

        
            function sumar(a, b, callback) {
                var resultado = a + b;
                callback(resultado);
            }

            sumar(5, 3, function(resultado) {
                console.log(resultado);
            });

Promise

Una Promise es un objeto que representa la eventual finalización o falla de una operación asíncrona. A continuación se muestra un ejemplo de cómo utilizar una Promise en javascript.

        
            function sumar(a, b) {
                return new Promise(function(resolve, reject) {
                    var resultado = a + b;
                    if (resultado >= 0) {
                        resolve(resultado);
                    } else {
                        reject("Error: El resultado es negativo");
                    }
                });
            }

            sumar(5, 3)
                .then(function(resultado) {
                    console.log(resultado);
                })
                .catch(function(error) {
                    console.log(error);
                });

IndexedDB

IndexedDB es una API de javascript que permite almacenar datos en el navegador de forma estructurada. A continuación se muestra un ejemplo de cómo utilizar IndexedDB en javascript.

        
            var request = indexedDB.open("mi-base-de-datos", 1);

            request.onupgradeneeded = function(event) {
                var db = event.target.result;
                var objectStore = db.createObjectStore("personas", { keyPath: "id" });
                objectStore.createIndex("nombre", "nombre", { unique: false });
                objectStore.createIndex("edad", "edad", { unique: false });
            };

            request.onsuccess = function(event) {
                var db = event.target.result;
                var transaction = db.transaction("personas", "readwrite");
                var objectStore = transaction.objectStore("personas");
                var request = objectStore.add({ id: 1, nombre: "Juan", edad: 25 });

                request.onsuccess = function(event) {
                    console.log("Persona agregada correctamente");
                };

                request.onerror = function(event) {
                    console.log("Error al agregar la persona");
                };
            };

            request.onerror = function(event) {
                console.log("Error al abrir la base de datos");
            };

Web SQL

Web SQL es una API de javascript que permite almacenar datos en el navegador de forma estructurada. A continuación se muestra un ejemplo de cómo utilizar Web SQL en javascript.

        
            var db = openDatabase("mi-base-de-datos", "1.0", "Mi base de datos", 2 * 1024 * 1024);

            db.transaction(function(tx) {
                tx.executeSql("CREATE TABLE IF NOT EXISTS personas (id unique, nombre, edad)");
                tx.executeSql("INSERT INTO personas (id, nombre, edad) VALUES (1, 'Juan', 25)");
            });

            db.transaction(function(tx) {
                tx.executeSql("SELECT * FROM personas", [], function(tx, results) {
                    var len = results.rows.length;
                    for (var i = 0; i < len; i++) {
                        var row = results.rows.item(i);
                        console.log(row.id + " " + row.nombre + " " + row.edad);
                    }
                });
            });

  • substring: Extrae los caracteres entre dos posiciones especificadas
  • toLowerCase: Convierte una cadena de texto a minúsculas
  • toUpperCase: Convierte una cadena de texto a mayúsculas
  • trim: Elimina los espacios en blanco al principio y al final de una cadena de texto
  • valueOf: Retorna el valor primitivo de un objeto String
  • startsWith: Determina si una cadena de texto comienza con los caracteres de otra cadena de texto
  • endsWith: Determina si una cadena de texto termina con los caracteres de otra cadena de texto
  • includes: Determina si una cadena de texto puede ser encontrada dentro de otra cadena de texto
  • repeat: Retorna una nueva cadena de texto que contiene el número especificado de copias de la cadena de texto original
  • padStart: Rellena la cadena de texto con un número específico de caracteres al principio
  • padEnd: Rellena la cadena de texto con un número específico de caracteres al final
  • trimStart: Elimina los espacios en blanco al principio de una cadena de texto
  • trimEnd: Elimina los espacios en blanco al final de una cadena de texto
  • normalize: Retorna la forma normalizada de una cadena de texto
  • codePointAt: Retorna el valor Unicode del carácter en la posición especificada
